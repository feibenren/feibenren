## 写个tab
```
<ul class="tabs">
    <li >tab1</li>
    <li>tab2</li>
    <li>tab3</li>
</ul>
<ul class="tab-contents">
    <li>tab1tab1tab1tab1</li>
    <li>tab2tab2tab2tab2</li>
    <li>tab3tab3tab3tab3</li>
</ul>
```
```
var tabs=document.getElementsByClassName('tabs')[0].getElementsByTagName('li');
var tabContents=document.getElementsByClassName('tab-contents')[0].getElementsByTagName('li');

// init
var index=0;
tabs[0].className+="active";
tabContents[0].className+="active";

for(var i=0;i<tabs.length;i++){
    tabs[i].dataset.id=i;
    tabs[i].addEventListener('click',function(ev){
        console.log(this.dataset.id);
        //清空tabs，tab-contents中li的active
        for(var i=0;i<tabs.length;i++){
            tabs[i].className=tabs[i].className.replace(/active/g,'');
        }
        for(var i=0;i<tabContents.length;i++){
            tabContents[i].className=tabContents[i].className.replace(/active/g,'');
        }
        this.className+='active';
        tabContents[this.dataset.id].className+='active';
    });
}
```
